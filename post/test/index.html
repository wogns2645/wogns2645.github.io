hugo + github

hugo 설치
brew install hugo

# Github repository 2개 생성 (blog, {username}.github.io)
# 프로젝트 생성
hugo new site blog

git init

# themes(https://themes.gohugo.io/)
# theme 경로 themes/theme 이름
git submodule add https://github.com/st-wong/hugo-spectre-pixel-theme.git themes/spectre-pixel

# config 복사
cp blog/themes/spectre-pixel/exampleSite/config.toml blog/config.toml

# config.toml 수정
# baseurl = "https://<username>.github.io/"

# 서버 실행 (localhost:1313)
hugo server

# Github blog repository 연결
git remote add origin https://{token}@github.com/wogns2645/blog.git

# Github {username}.github.io repository 연결
git submodule add -b master https://github.com/wogns2645/wogns2645.github.io.git public

# 배포
# blog/deploy.sh 루트 디렉토리에 생성
# 권한 / 실행
chmod 777 blog/deploy.sh
./deploy.sh

# 게시글 생성
hugo new post/blod.md
git remote/submodule 초기화
git remote remove origin
rm -rf public
rm -rf .git/modules/{submodule}

# Already exists in the index 문제
# 캐쉬 확인
git ls-files --stage {폴더 경로}
# 캐시 삭제
git rm --cached {폴더 경로}
deploy.sh
#!/bin/bash

echo -e "\033[0;32mDeploying updates to GitHub...\033[0m"

# Build the project.
# hugo -t 테마명
hugo -t spectre-pixel

# Go To Public folder, sub module commit
cd public
# Add changes to git.
git add .

# Commit changes.
msg="rebuilding site `date`"
if [ $# -eq 1 ]
  then msg="$1"
fi
git commit -m "$msg"

# Push source and build repos.
git push origin master

# Come Back up to the Project Root
cd ..


# blog 저장소 Commit & Push
git add .

msg="rebuilding site `date`"
if [ $# -eq 1 ]
  then msg="$1"
fi
git commit -m "$msg"

git push origin master
